create table users(
   id INT NOT NULL AUTO_INCREMENT,
   appID INT NOT NULL,
   email VARCHAR(120),
   firstname VARCHAR(120),
   lastname VARCHAR(120),
   password VARCHAR(80) NOT NULL UNIQUE,
   birthdate VARCHAR(120),
   gender VARCHAR(120),
   address VARCHAR(120),
   city VARCHAR(120),
   state VARCHAR(120),
   zipCode VARCHAR(120),
   phone VARCHAR(120),
   admin INT NOT NULL DEFAULT 0,
   PRIMARY KEY ( id )
);
